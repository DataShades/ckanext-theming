{%- macro alert(content, dismissible, style="info") -%} {%- set style_class = style if style.startswith("alert-") else "alert-" ~ style -%} {%- set class = "alert fade show " ~ ("alert-dismissible " if dismissible else "") ~ style_class -%}
{{ content }} {%- if dismissible -%} {%- endif %}
{%- endmacro %} {%- macro modal(content, title, footer, id, size, dismissible=true) -%} {%- set class = "modal fade" -%} {%- if size -%} {%- set class = class ~ " modal-" ~ size -%} {%- endif %} {%- set default_attrs = {"class": class, "id": id, "tabindex": "-1"} -%} {%- if not dismissible -%} {%- do default_attrs.update({"data-bs-backdrop": "static", "data-bs-keyboard": "false"}) -%} {%- endif %}
{%- if title or dismissible -%}
{%- if title -%}
{{ title }}
{%- endif %} {%- if dismissible -%} {%- endif %}
{%- endif %}
{{ content }}
{%- if footer -%}
{{ footer }}
{%- endif %}
{%- endmacro %} {%- macro popover(content, title, id) -%}
{{ content }}
{%- endmacro %} {%- macro progress(content, value=0, max=100) -%}
{%- if content %}{{ content }}{% else %}{{ value }} / {{ max }}{% endif %}
{%- endmacro %} {%- macro spinner(content, style, size, variant="border") -%} {%- set class = "spinner-" ~ variant -%} {%- if size -%} {%- set class = "{} {}-{}".format(class, class, size) -%} {%- endif %} {%- if style -%} {%- set class = class ~ " text-" ~style -%} {%- endif %} {{ content or _("Loading...") }} {%- endmacro %} {%- macro toast(content, title, style, dismissible) -%} {%- set class = "toast fade show" -%} {%- if style -%} {%- set class = class ~ " text-bg-" ~ style -%} {%- endif %}
{%- if title or dismissible -%}
{{ title }} {%- if dismissible -%} {%- endif %}
{%- endif %}
{{ content }}
{%- endmacro %} {%- macro toast_stack(content, position="bottom-right") -%} {%- set class = "toast-container position-fixed m-2" -%} {%- set class = class ~ (" top-0" if position.startswith("top") else " bottom-0") -%} {%- set class = class ~ (" start-0" if position.startswith("left") else " end-0") -%}
{{- content -}}
{%- endmacro %} {%- macro tooltip(content, tooltip, position) -%} {{- content -}} {%- endmacro %}
